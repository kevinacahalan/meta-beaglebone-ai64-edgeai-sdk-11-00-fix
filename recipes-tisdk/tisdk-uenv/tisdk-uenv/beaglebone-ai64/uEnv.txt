# uEnv.txt for BeagleBone AI-64 (TISDK 11.00 / edgeai image), SD-card boot.
#
# Why this file exists:
#   The stock meta-arago uEnv.txt is empty, so the BeagleBoard.org U-Boot
#   (u-boot-bb.org) runs "bootflow scan", finds the eMMC's extlinux.conf and
#   boots eMMC instead of this SD card. By setting "uenvcmd", U-Boot's "envboot"
#   runs it (before the bootflow scan) and boots the TI way from the SD card.
#
#   bootcmd_ti_mmc loads ${bootdir}/${name_kern} (=/boot/Image) and
#   ${bootdir}/dtb/${fdtfile} from ${bootpart}. The kernel + dtb live on the
#   ext4 rootfs (partition 2), so bootpart must be 1:2, not the default 1:1
#   (the FAT boot partition has no kernel).

bootpart=1:2
fdtfile=ti/k3-j721e-beagleboneai64.dtb
uenvcmd=run bootcmd_ti_mmc
